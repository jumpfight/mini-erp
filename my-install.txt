1. Setup Backend Go
mini-erp/
├── backend/
│   ├── main.go
│   ├── routes/
│   │   └── user.go
│   └── models/
│       └── user.go

Modul dan Dependensi
go mod init mini-erp
go get github.com/gorilla/mux
go get github.com/rs/cors


2. Setup Frontend ReactJS
frontend/
├── src/
│   ├── App.jsx
│   └── components/UserList.jsx

npm create vite@latest frontend --template react
cd frontend
npm install

3. Setup docker

    - backend
    FROM golang:1.21-alpine

    # Buat folder kerja
    WORKDIR /app

    # Copy file project ke container
    COPY . .

    # Download dependensi
    RUN go mod download

    # Build binary
    RUN go build -o main .

    # Jalankan binary
    CMD ["./main"]

    # Port backend (ubah sesuai yg kamu pakai di main.go)
    EXPOSE 8080

    - frontend
    # Tahap 1: Build React App
    FROM node:18-alpine AS builder

    WORKDIR /app
    COPY . .

    RUN npm install
    RUN npm run build

    # Tahap 2: Serve pakai nginx
    FROM nginx:alpine

    # Copy hasil build React ke nginx
    COPY --from=builder /app/dist /usr/share/nginx/html

    # Copy config nginx opsional
    COPY nginx.conf /etc/nginx/conf.d/default.conf

    EXPOSE 80


jalankan dockter desktop 
docker-compose up --build
